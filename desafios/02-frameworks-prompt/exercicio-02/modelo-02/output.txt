#!/bin/bash
# ==============================================================================
# Script de Backup Automatizado do PostgreSQL
# ==============================================================================
# Descrição: Realiza o dump do banco PostgreSQL, compacta com gzip, envia para 
#            o Amazon S3, aplica retenção de 30 dias e registra logs com timestamp.
# 
# Host: ledger-db.internal.hvt.io
# Banco: ledger_prod
# Destino S3: s3://hvt-ledger-backups/
# SO: Ubuntu 22.04 LTS
# ==============================================================================

# Interrompe a execução imediatamente se qualquer comando falhar
set -eo pipefail

# ==============================================================================
# Variáveis Configuráveis
# ==============================================================================
DB_HOST="ledger-db.internal.hvt.io"
DB_PORT="5432"
DB_NAME="ledger_prod"
DB_USER="backup_user"

BACKUP_DIR="/var/backups/ledger"
S3_BUCKET="s3://hvt-ledger-backups"
AWS_REGION="us-east-1"
RETENTION_DAYS=30

LOG_FILE="/var/log/ledger-backup.log"

# Data e Hora para o nome do arquivo de backup
TIMESTAMP=$(date +"%Y%m%d_%H%M%S")
BACKUP_FILENAME="${DB_NAME}_${TIMESTAMP}.sql.gz"
LOCAL_BACKUP_PATH="${BACKUP_DIR}/${BACKUP_FILENAME}"

# ==============================================================================
# Função de Log
# ==============================================================================
log() {
    local MESSAGE="[$(date '+%Y-%m-%d %H:%M:%S')] $1"
    echo "$MESSAGE" | tee -a "$LOG_FILE"
}

# ==============================================================================
# Início do Processo
# ==============================================================================
log "INFO: Iniciando o processo de backup para o banco ${DB_NAME}..."

# Garante que o diretório de backup existe
mkdir -p "$BACKUP_DIR"

# 1. Execução do pg_dump compactado diretamente via gzip
log "INFO: Executando pg_dump e compactando com gzip..."
if pg_dump -h "$DB_HOST" -p "$DB_PORT" -U "$DB_USER" -d "$DB_NAME" -F c | gzip > "$LOCAL_BACKUP_PATH"; then
    log "SUCESSO: Dump e compactação concluídos: ${LOCAL_BACKUP_PATH}"
else
    log "ERRO: Falha crítica durante a execução do pg_dump."
    exit 1
fi

# 2. Upload para o S3
log "INFO: Enviando o arquivo para o S3 (${S3_BUCKET}/${BACKUP_FILENAME})..."
if aws s3 cp "$LOCAL_BACKUP_PATH" "${S3_BUCKET}/${BACKUP_FILENAME}" --region "$AWS_REGION"; then
    log "SUCESSO: Upload para o S3 concluído com sucesso."
else
    log "ERRO: Falha ao enviar o arquivo de backup para o S3."
    rm -f "$LOCAL_BACKUP_PATH"
    exit 1
fi

# 3. Limpeza do arquivo local para liberar espaço no disco
log "INFO: Removendo o arquivo local temporário..."
rm -f "$LOCAL_BACKUP_PATH"

# 4. Política de Retenção no S3 (Remover backups com mais de X dias)
log "INFO: Aplicando política de retenção de ${RETENTION_DAYS} dias no S3..."

# Lista e remove arquivos antigos com base na retenção configurada
CUTOFF_DATE=$(date -d "${RETENTION_DAYS} days ago" +%s)

aws s3 ls "${S3_BUCKET}/" --region "$AWS_REGION" | while read -r line; do
    # Extrai a data/hora e o nome do arquivo do log do S3 ls
    file_date=$(echo "$line" | awk '{print $1" "$2}')
    file_name=$(echo "$line" | awk '{print $4}')
    
    # Verifica se a linha contém um arquivo de backup válido do banco
    if [[ "$file_name" =~ ^${DB_NAME}_[0-9]{8}_[0-9]{6}\.sql\.gz$ ]]; then
        file_timestamp=$(date -d "$file_date" +%s 2>/dev/null || echo 0)
        
        if (( file_timestamp < CUTOFF_DATE )); then
            log "INFO: Removendo arquivo expirado do S3: ${file_name}"
            aws s3 rm "${S3_BUCKET}/${file_name}" --region "$AWS_REGION"
        fi
    fi
done

log "SUCESSO: Rotina de backup finalizada com sucesso."
exit 0
