SELECT 
    to_char(completed_at, 'YYYY-MM') AS mes,
    category AS categoria,
    COUNT(id) AS quantidade_transacoes,
    ROUND(SUM(amount_cents) / 100.0, 2) AS volume_total_brl
FROM 
    transactions
WHERE 
    status = 'completed'
    AND category IN ('subscription', 'one_time', 'refund', 'credit_adjustment')
    -- Recorte de 6 meses corridos considerando o limite final às 23:59:59 do dia 24/04/2026
    AND completed_at >= '2026-04-24 00:00:00'::timestamp - INTERVAL '6 months'
    AND completed_at <= '2026-04-24 23:59:59'::timestamp
GROUP BY 
    to_char(completed_at, 'YYYY-MM'),
    category
ORDER BY 
    mes ASC,
    categoria ASC;
